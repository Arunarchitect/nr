server {
    server_name modelflick.com www.modelflick.com;

    location = /favicon.ico { access_log off; log_not_found off; }

    location / {
        include proxy_params;
        proxy_pass http://unix:/run/gunicorn.sock;

        add_header 'Access-Control-Allow-Origin' 'https://www.modelflick.com' always;
    }

    location  /static/ {
        root /home/django/nr;
    }

    location  /media/ {
        root /home/django/nr;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/modelflick.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/modelflick.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}


server {
    if ($host = www.modelflick.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = modelflick.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name modelflick.com www.modelflick.com;
    return 404; # managed by Certbot




}
